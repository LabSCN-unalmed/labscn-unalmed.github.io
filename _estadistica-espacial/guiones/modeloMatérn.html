Hallar los valores de phi y de kappa, para un rango pr√°ctico fijo

# Funci√≥n de Mat√©rn para igualar a 0
# Nivel de rango pr√°ctico rho = 0.05

require(geoR)
fM <- function(x,kappa, rho = 0.05, phi = 1) 
  cov.spatial(x,cov.pars=c(1,phi),kappa=kappa,cov.model="matern")-rho

# Hallar los l√≠mites para encontrar la raiz de la funci√≥n
fM(c(2,11), kappa=0.5)
## [1]  0.08533528 -0.04998330
fM(c(2,11), kappa=2.5)
## [1]  0.53645289 -0.04912594
# Encontrar el factor multiplicador de phi
(r1 <- uniroot(fM,c(2,11),kappa=0.5)$root)
## [1] 2.995747
(r2 <- uniroot(fM,c(2,11),kappa=1)$root)
## [1] 3.998538
(r3 <- uniroot(fM,c(2,11),kappa=1.5)$root)
## [1] 4.743862
(r4 <- uniroot(fM,c(2,11),kappa=2.5)$root)
## [1] 5.918654
(r5 <- uniroot(fM,c(2,11),kappa=3)$root)
## [1] 6.417135
# Dado un rango pr√°ctico
RangoP <- 0.75
