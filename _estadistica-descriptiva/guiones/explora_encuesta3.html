encuesta <- encuesta %>%
            mutate(Hora_lleg = HoraD + MinD/60,
                   Hora_sal  = HoraO + MinO/60,
                   Tmpo_viaje = Hora_lleg - Hora_sal,
                   puntualidad = if_else(Hora_lleg < 7,

                                         (Hora_lleg - 8) * 60))
