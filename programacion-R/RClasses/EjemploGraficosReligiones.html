Clculo del porcentaje de cada ingreso por religin.

religT1 <- aggregate(subset(relig2, select=n), 
          with(relig2, list(Religin = Religin)), sum)
names(religT1)[2]<-"nT"
head(relig2)
##             Religin Ingresos   n
## 1           Agnostic    <$10k  27
## 2            Atheist    <$10k  12
## 3           Buddhist    <$10k  27
## 4           Catholic    <$10k 418
## 5 Dont know/refused    <$10k  15
## 6   Evangelical Prot    <$10k 575
relig2 <- merge(relig2, religT1)
