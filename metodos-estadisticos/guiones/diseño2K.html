# Funciones auxiliares para hallar el error est√°ndar de los resdiuales
# de un modelo "aov"
errsdred<-function(modelo){
  aovm<-anova(modelo)
  return(sqrt(aovm$"Mean Sq"[length(aovm$"Mean Sq")]))
}

efectosignf<-function(modelo,alfa=0.05){
  aovm<-anova(modelo)
  tm<-model.tables(modelo,se=T)
  glmod<-aovm$Df
  corrm<-sqrt(glmod[length(glmod)]/glmod[1:(length(glmod)-1)])
  cat("\nEfectos ajustados\n")
  print(lapply(1:length(tm$tables),function(i,a,b) a$tables[[i]]*b[i],tm,corrm))
  seRes<-sqrt(aovm$"Mean Sq"[length(aovm$"Mean Sq")])
  cat("\nNivel de comparaci√≥n\n")
  print(seRes*qt(1-alfa/2,glmod[length(glmod)]))
}
